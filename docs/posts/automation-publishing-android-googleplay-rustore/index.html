tasks.register("bumpVersion") {
    group = "versioning"
    description = "Bump VERSION_CODE/NAME. -Ppart=patch|minor|major or -PnewName=1.2.3"

    doLast {
        val propsFile = file("${rootDir}/gradle.properties")
        val props = Properties().apply { propsFile.inputStream().use { load(it) } }

        val code = (props.getProperty("VERSION_CODE") ?: "1").toInt()
        val name = props.getProperty("VERSION_NAME") ?: "0.1.0"

        val newName = (project.findProperty("newName") as String?) ?: bumpSemver(
            name,
            project.findProperty("part") as String?
        )

        props.setProperty("VERSION_CODE", (code + 1).toString())
        props.setProperty("VERSION_NAME", newName)
        propsFile.outputStream().use { props.store(it, null) }

        println("Bumped: VERSION_CODE $code -> ${code + 1}, VERSION_NAME $name -> $newName")
    }
}

tasks.register("tagVersion") {
    group = "versioning"
    dependsOn("bumpVersion")
    doLast {
        val props = Properties().apply {
            file("${rootDir}/gradle.properties").inputStream().use { load(it) }
        }
        val v = props.getProperty("VERSION_NAME")
        exec { commandLine("git", "add", "gradle.properties") }
        exec { commandLine("git", "commit", "-m", "chore: bump version to $v") }
        exec { commandLine("git", "tag", "-a", "v$v", "-m", "release v$v") }
        exec { commandLine("git", "push", "origin", "HEAD") }
        exec { commandLine("git", "push", "origin", "v$v") }
        println("Tagged and pushed v$v")
    }
}
